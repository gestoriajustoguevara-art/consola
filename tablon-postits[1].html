📌 Necesitas crear una hoja de Google Sheets con el nombre PostIts y desplegar un Apps Script. Pulsa el botón de abajo para ver las instrucciones paso a paso.


Paso 1: Abre Google Sheets y crea una hoja nueva.
Paso 2: Renombra la pestaña como PostIts.
Paso 3: En el menú: Extensiones → Apps Script.
Paso 4: Borra el contenido y pega el código que tienes más abajo.
Paso 5: Haz clic en Implementar → Nueva implementación.
   - Tipo: Aplicación web
   - Ejecutar como: Yo
   - Acceso: Cualquier persona
Paso 6: Copia la URL que te da y pégala arriba en "URL del Apps Script".

📋 Código Apps Script: 
function doGet(e) {
  return handleRequest(e);
}
function doPost(e) {
  return handleRequest(e);
}
function handleRequest(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("PostIts");
  var action = e.parameter.action || (e.postData ? JSON.parse(e.postData.contents).action : '');
  var data = e.parameter;
  if (e.postData) {
    try { data = JSON.parse(e.postData.contents); } catch(ex) {}
  }
  var result = {};
  if (action === 'get') {
    var rows = sheet.getDataRange().getValues();
    var notas = [];
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][7] !== 'ELIMINADA') {
        notas.push({
          id: rows[i][0], de: rows[i][1], para: rows[i][2],
          asunto: rows[i][3], texto: rows[i][4], color: rows[i][5],
          fecha: rows[i][6], estado: rows[i][7]
        });
      }
    }
    result = { notas: notas };
  } else if (action === 'post') {
    var id = new Date().getTime().toString();
    sheet.appendRow([id, data.de, data.para, data.asunto, data.texto, data.color, data.fecha, 'activa']);
    result = { ok: true, id: id };
  } else if (action === 'delete') {
    var rows2 = sheet.getDataRange().getValues();
    for (var j = 1; j < rows2.length; j++) {
      if (rows2[j][0] == data.id) {
        sheet.getRange(j+1, 8).setValue('ELIMINADA');
        break;
      }
    }
    result = { ok: true };
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
